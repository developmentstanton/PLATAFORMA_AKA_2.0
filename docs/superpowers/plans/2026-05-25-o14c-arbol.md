# O14C Árbol jerárquico — Plan de Implementación

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convertir la pestaña O14C en un árbol colapsable Grupo → Almacén → Negocio (todos los negocios), con CEDI como grupo aparte, subtotales en cliente y filtro por negocio sin refetch.

**Architecture:** El endpoint (`api/informe_o14.php`, `tab=c`) arma una jerarquía anidada `grupos[]→almacenes[]→negocios[]→valores[medida][talla]` desde `#base` (que ya incluye ventas). El cliente (`informes/o14.php`) calcula subtotales y pinta una tabla en árbol con filas mostrables/ocultables vía re-render. Sin librerías (tidy-data + pivote en cliente).

**Tech Stack:** PHP + sqlsrv (SQL Server), JS vanilla, SweetAlert2 (ya presente). Pruebas de endpoint vía harness CLI contra BD real (patrón existente del proyecto).

**Nota de commits:** todo plataforma_20 está untracked por decisión de Rafael; este plan usa pasos de **verificación** en vez de commits. Los commits se harán en lote cuando Rafael lo pida.

/** Arma la jerarquía grupos→almacenes→negocios desde filas planas de #base+Bodegas.
 *  KPIs de cantidad se acumulan EXCLUYENDO el grupo CEDI. */
function ensamblarArbol($rows) {
    $MED = ['siembra','disponible','hold','disphold','sobrante','faltante','ventas'];
    $tallasSet=[]; $arbol=[];
    $kpi=['siembra'=>0,'disponible'=>0,'hold'=>0,'ventas'=>0,'sobrantes'=>0,'faltante'=>0];
    foreach ($rows as $r) {
        $g=$r['grupo']; $ll=$r['llave']; $neg=$r['negocio']; $talla=(string)$r['talla']; $tallasSet[$talla]=true;
        $si=(int)$r['siembra']; $di=(int)$r['disponible']; $ho=(int)$r['hold']; $ve=(int)$r['ventas'];
        $bal=$si-($di+$ho); $fal=max(0,$bal); $sob=max(0,-$bal);
        if(!isset($arbol[$g])) $arbol[$g]=['grupo'=>$g,'almacenes'=>[]];
        if(!isset($arbol[$g]['almacenes'][$ll])) $arbol[$g]['almacenes'][$ll]=['llave'=>$ll,'bodega'=>$r['bodega'],'nombre'=>$r['nombre'],'negocios'=>[]];
        if(!isset($arbol[$g]['almacenes'][$ll]['negocios'][$neg])) $arbol[$g]['almacenes'][$ll]['negocios'][$neg]=['negocio'=>$neg,'referencia'=>$r['referencia'],'color'=>$r['color'],'valores'=>[]];
        $vals=&$arbol[$g]['almacenes'][$ll]['negocios'][$neg]['valores'];
        foreach(['siembra'=>$si,'disponible'=>$di,'hold'=>$ho,'disphold'=>$di+$ho,'sobrante'=>$sob,'faltante'=>$fal,'ventas'=>$ve] as $m=>$v)
            $vals[$m][$talla]=($vals[$m][$talla]??0)+$v;
        unset($vals);
        if($g!=='CEDI'){ $kpi['siembra']+=$si; $kpi['disponible']+=$di; $kpi['hold']+=$ho; $kpi['ventas']+=$ve; $kpi['sobrantes']+=$sob; $kpi['faltante']+=$fal; }
    }
    $grupos=[];
    foreach($arbol as $g){
        $g['almacenes']=array_values(array_map(function($a){ $a['negocios']=array_values($a['negocios']); return $a; }, $g['almacenes']));
        $grupos[]=$g;
    }
    $tallas=array_keys($tallasSet);
    usort($tallas, fn($a,$b)=>(is_numeric($a)&&is_numeric($b))?($a<=>$b):strcmp($a,$b));
    return [$grupos, $tallas, $kpi];
}

/** KPIs de conteo (red, EXCLUYENDO CEDI) desde #base. */
function kpiCounts($c) {
    $r = run($c, "
        SELECT
          (SELECT COUNT(DISTINCT cia+'|'+negocio) FROM #base WHERE bodega<>'CEDI')              negocios,
          (SELECT COUNT(DISTINCT cia+'|'+negocio) FROM #base WHERE bodega<>'CEDI' AND siembra>0) negocios_con_siembra,
          (SELECT COUNT(DISTINCT cia+'-'+bodega)   FROM #base WHERE bodega<>'CEDI' AND siembra>0) tiendas_con_siembra,
          (SELECT COUNT(DISTINCT cia+'-'+bodega)   FROM #base WHERE bodega<>'CEDI' AND disponible>0) tiendas_con_inv,
          (SELECT COUNT(DISTINCT cia+'-'+bodega)   FROM #base WHERE bodega<>'CEDI' AND ventas<>0) tiendas_con_venta");
    if (isset($r['error']) || !$r) return [];
    return array_map('intval', $r[0]);
}
```

  // Suma valores[medida][talla] sobre una lista de negocios (para subtotales de grupo/almacén).
  function sumValores(negocios, medidas){
    const out={}; medidas.forEach(m=>out[m]={});
    negocios.forEach(n=>medidas.forEach(m=>{ const v=n.valores[m]||{}; for(const t in v) out[m][t]=(out[m][t]||0)+v[t]; }));
    return out;
  }
  // Celdas de medidas×tallas+Tot para una fila. colorize=true solo en hojas (negocio).
  function rowCells(valores, tallas, medidas, colorize){
    let h='';
    medidas.forEach(m=>{ let tot=0; tallas.forEach(t=>{ const v=(valores[m]||{})[t]||0; tot+=v;
      const cls=colorize?((m==='faltante'&&v>0)?' class="o14-cell-fal"':((m==='sobrante'&&v>0)?' class="o14-cell-sob"':'')):'';
      h+='<td'+cls+'>'+(v?nf(v):'')+'</td>'; }); h+='<td class="blocktot">'+nf(tot)+'</td>'; });
    return h;
  }

  // Poda el árbol a un solo negocio (ref+color), descartando almacenes/grupos vacíos.
  function filterArbol(data, ref, color){
    const grupos=[];
    (data.grupos||[]).forEach(g=>{
      const alm=[];
      g.almacenes.forEach(a=>{ const negs=a.negocios.filter(n=>n.referencia===ref && n.color===color);
        if(negs.length) alm.push(Object.assign({}, a, {negocios:negs})); });
      if(alm.length) grupos.push(Object.assign({}, g, {almacenes:alm}));
    });
    return Object.assign({}, data, {grupos});
  }
```
