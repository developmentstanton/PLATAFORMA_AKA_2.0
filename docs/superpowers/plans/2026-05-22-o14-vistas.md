# O14 — Endpoint + Vistas (O14B / O14C / Recomendaciones) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Construir el informe nativo O14 (matrices Siembra/Disponible/Hold/Ventas/Sobrante/Faltante por talla, a nivel negocio y tienda) y el tab de Recomendaciones que consume el motor ya construido.

**Tech Stack:** PHP 8 + sqlsrv (SQL Server), JS vanilla + SweetAlert2 (CDN, ya cargado en dashboard.php). Sin npm/build. Tests de endpoint vía harness PHP CLI (patrón G00).

**Spec:** `docs/superpowers/specs/2026-05-22-o14-design.md`. **Motor + tests:** `api/o14_recomendador.php`, `tests/o14_recomendador_test.php`.

- [ ] **Step 1: Crear `api/lib_refs.php`** copiando las dos funciones tal cual están en `api/informe_g00.php` (`getRefsCached($conn,$proveedor)` y `buildRefsTemp($conn,$refs)`), envueltas en `if (!function_exists(...))` para evitar choques si en el futuro se incluye junto a G00.

```php
<?php
/** Helpers compartidos para materializar las referencias del proveedor en #refs. */
if (!function_exists('getRefsCached')) {
    function getRefsCached($conn, $proveedor) {
        $cacheDir = __DIR__ . '/../cache';
        if (!is_dir($cacheDir)) @mkdir($cacheDir, 0755, true);
        $cacheFile = $cacheDir . '/g00_refs_' . md5($proveedor) . '.json';
        if (file_exists($cacheFile) && date('Y-m-d', filemtime($cacheFile)) === date('Y-m-d')) {
            $data = json_decode(file_get_contents($cacheFile), true);
            if (is_array($data)) return $data;
        }
        $sql = "SELECT REFERENCIA,
                    ISNULL(MARCA,'SIN MARCA') AS MARCA, ISNULL(LINEA,'SIN LINEA') AS LINEA,
                    ISNULL(SUBLINEA,'') AS SUBLINEA, ISNULL(CATEGORIA,'') AS CATEGORIA
                FROM INTEGRACION.dbo.ITEMS WITH (NOLOCK) WHERE PROVEEDOR = ?";
        $stmt = sqlsrv_query($conn, $sql, [$proveedor]);
        if ($stmt === false) return [];
        $rows = [];
        while ($r = sqlsrv_fetch_array($stmt, SQLSRV_FETCH_ASSOC)) $rows[] = $r;
        sqlsrv_free_stmt($stmt);
        @file_put_contents($cacheFile, json_encode($rows));
        return $rows;
    }
    function buildRefsTemp($conn, $refs) {
        $ok = sqlsrv_query($conn, "CREATE TABLE #refs (
            REFERENCIA varchar(50) NOT NULL PRIMARY KEY,
            MARCA varchar(40), LINEA varchar(40), SUBLINEA varchar(40), CATEGORIA varchar(40))");
        if ($ok === false) return false;
        sqlsrv_free_stmt($ok);
        if (empty($refs)) return true;
        foreach (array_chunk($refs, 200) as $chunk) {
            $vals = []; $params = [];
            foreach ($chunk as $r) {
                $vals[] = '(?,?,?,?,?)';
                array_push($params, $r['REFERENCIA'], $r['MARCA'], $r['LINEA'], $r['SUBLINEA'], $r['CATEGORIA']);
            }
            $ins = sqlsrv_query($conn, "INSERT INTO #refs (REFERENCIA,MARCA,LINEA,SUBLINEA,CATEGORIA) VALUES " . implode(',', $vals), $params);
            if ($ins === false) return false;
            sqlsrv_free_stmt($ins);
        }
        return true;
    }
}
```

- [ ] **Step 2: Construir `#base` (siembra ⟕ disponible ⟕ hold por llave)**

Agregar al endpoint. Se hace UNION de llaves de las 3 fuentes y se agrega cada cantidad con `SUM` (defensivo ante duplicados). `@cia` opcional.

```php
$filtroCia = $cia !== '' ? " AND CIA = ? " : "";
$pCia = $cia !== '' ? [$cia] : [];

$sqlBase = "
  CREATE TABLE #base (cia varchar(10), bodega varchar(20), negocio varchar(80),
                      referencia varchar(50), color varchar(40), talla varchar(40),
                      siembra int, disponible int, hold int);
  WITH s AS (
    SELECT rtrim(f400_id_cia) cia, rtrim(f150_id) bodega, rtrim(f120_referencia) referencia,
           rtrim(f121_id_ext1_detalle) color, rtrim(f121_id_ext2_detalle) talla,
           SUM(f400_cant_nivel_min_1) q
    FROM stanton.dbo.t400_cm_existencia
     INNER JOIN stanton.dbo.t150_mc_bodegas ON f150_rowid=f400_rowid_bodega
     INNER JOIN stanton.dbo.t121_mc_items_extensiones ON f121_rowid=f400_rowid_item_ext
     INNER JOIN stanton.dbo.t120_mc_items ON f120_rowid=f121_rowid_item
     INNER JOIN #refs r ON r.REFERENCIA = rtrim(f120_referencia)
    WHERE (f400_cant_nivel_min_1>0 OR f400_cant_nivel_pedido>0) AND f120_referencia<>'GIFTCARD'
    GROUP BY rtrim(f400_id_cia),rtrim(f150_id),rtrim(f120_referencia),rtrim(f121_id_ext1_detalle),rtrim(f121_id_ext2_detalle)
  ),
  d AS (
    SELECT rtrim(cia) cia, rtrim(bodega) bodega, rtrim(referencia) referencia,
           rtrim(color) color, rtrim(talla) talla, SUM(cantidad) q
    FROM INTEGRACION.dbo.inv_actual_PBI v WITH (NOLOCK)
     INNER JOIN #refs r ON r.REFERENCIA = rtrim(v.referencia)
    WHERE cia<>'001' AND COLUMNA1 IN ('INV1430','INV1435','400')
    GROUP BY rtrim(cia),rtrim(bodega),rtrim(referencia),rtrim(color),rtrim(talla)
  ),
  h AS (
    SELECT rtrim(cia) cia, rtrim(bodega_ent) bodega, rtrim(referencia) referencia,
           rtrim(color) color, rtrim(talla) talla, SUM(cantidad) q
    FROM INTEGRACION.dbo._hold_actual_PBI v WITH (NOLOCK)
     INNER JOIN #refs r ON r.REFERENCIA = rtrim(v.referencia)
    WHERE cia<>'001'
    GROUP BY rtrim(cia),rtrim(bodega_ent),rtrim(referencia),rtrim(color),rtrim(talla)
  ),
  llaves AS (
    SELECT cia,bodega,referencia,color,talla FROM s
    UNION SELECT cia,bodega,referencia,color,talla FROM d
    UNION SELECT cia,bodega,referencia,color,talla FROM h
  )
  INSERT INTO #base
  SELECT k.cia, k.bodega, k.referencia + '-' + k.color, k.referencia, k.color, k.talla,
         ISNULL(s.q,0), ISNULL(d.q,0), ISNULL(h.q,0)
  FROM llaves k
   LEFT JOIN s ON s.cia=k.cia AND s.bodega=k.bodega AND s.referencia=k.referencia AND s.color=k.color AND s.talla=k.talla
   LEFT JOIN d ON d.cia=k.cia AND d.bodega=k.bodega AND d.referencia=k.referencia AND d.color=k.color AND d.talla=k.talla
   LEFT JOIN h ON h.cia=k.cia AND h.bodega=k.bodega AND h.referencia=k.referencia AND h.color=k.color AND h.talla=k.talla
  WHERE 1=1 $filtroCia;
";
$mk = run($dbConnect, $sqlBase, $pCia);
if (isset($mk['error'])) jsonFail($mk, $dbConnect);
```

> Nota de ejecución: `CREATE TABLE #base` sin params + el `INSERT...SELECT` con `?` (filtroCia) en el mismo batch funciona porque `#base` ya existe en el scope antes del `INSERT` (mismo patrón/gotcha que `#refs`). Si SQL Server se queja por el `?` en el batch combinado, separar el `CREATE TABLE` en un `sqlsrv_query` aparte y dejar el `INSERT` en otro.

```php
function ensamblarTidy($agg, $ventas, $keyField) {
    // index ventas por negocio|talla
    $vmap = [];
    foreach ($ventas as $r) $vmap[$r['negocio'].'|'.$r['talla']] = (int)$r['ventas'];
    $tallasSet = [];
    $filas = [];   // key => ['key'=>..,'valores'=>[medida=>[talla=>n]]]
    $kpi = ['siembra'=>0,'disponible'=>0,'ventas'=>0,'sobrantes'=>0,'faltante'=>0];
    foreach ($agg as $r) {
        $k = $r[$keyField];
        $talla = (string)$r['talla']; $tallasSet[$talla] = true;
        $si=(int)$r['siembra']; $di=(int)$r['disponible']; $ho=(int)$r['hold'];
        $ve = $vmap[$r['negocio'].'|'.$talla] ?? 0;
        $bal = $si - ($di + $ho);
        $fal = max(0,$bal); $sob = max(0,-$bal);
        if (!isset($filas[$k])) $filas[$k] = ['key'=>$r, 'valores'=>[]];
        foreach (['siembra'=>$si,'disponible'=>$di,'hold'=>$ho,'ventas'=>$ve,'sobrante'=>$sob,'faltante'=>$fal] as $m=>$val) {
            $filas[$k]['valores'][$m][$talla] = ($filas[$k]['valores'][$m][$talla] ?? 0) + $val;
        }
        $kpi['siembra']+=$si; $kpi['disponible']+=$di; $kpi['ventas']+=$ve;
        $kpi['sobrantes']+=$sob; $kpi['faltante']+=$fal;
    }
    // orden natural de tallas (numérico cuando aplica)
    $tallas = array_keys($tallasSet);
    usort($tallas, fn($a,$b)=> (is_numeric($a)&&is_numeric($b)) ? ($a<=>$b) : strcmp($a,$b));
    return [array_values($filas), $tallas, $kpi];
}

[$filas, $tallas, $kpi] = ensamblarTidy($agg, $ventas, 'negocio');
```

```js
function renderMatriz(containerId, data, dimLabel) {
    const cont = document.getElementById(containerId);
    if (!data.filas || !data.filas.length) { cont.innerHTML = '<p style="padding:16px;color:var(--text-light)">Sin datos</p>'; return; }
    const tallas = data.tallas, medidas = data.medidas;
    const nf = (n)=> (n||0).toLocaleString('es-CO');
    // doble header: bloque de medida (colspan tallas+1) / tallas + Total
    let h = '<div class="o14-matriz"><table><thead><tr><th class="dim" rowspan="2">'+esc(dimLabel)+'</th>';
    medidas.forEach(m => h += '<th colspan="'+(tallas.length+1)+'">'+esc(m.toUpperCase())+'</th>');
    h += '</tr><tr>';
    medidas.forEach(()=> { tallas.forEach(t=> h+='<th>'+esc(t)+'</th>'); h+='<th>Tot</th>'; });
    h += '</tr></thead><tbody>';
    data.filas.forEach(fila => {
        const etiqueta = fila.key.nombre ? (esc(fila.key.bodega)+' '+esc(fila.key.nombre)) : esc(fila.key.negocio);
        h += '<tr><td class="dim">'+etiqueta+'</td>';
        medidas.forEach(m => {
            let tot=0;
            tallas.forEach(t => {
                const v = (fila.valores[m]||{})[t] || 0; tot+=v;
                const cls = m==='faltante'&&v>0 ? ' class="o14-cell-faltante"' : (m==='sobrante'&&v>0 ? ' class="o14-cell-sobrante"' : '');
                h += '<td'+cls+'>'+ (v?nf(v):'') +'</td>';
            });
            h += '<td><strong>'+nf(tot)+'</strong></td>';
        });
        h += '</tr>';
    });
    h += '</tbody></table></div>';
    cont.innerHTML = h;
}
function esc(s){ return (s==null?'':String(s)).replace(/[&<>"']/g,c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'})[c]); }
```
